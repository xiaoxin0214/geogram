471  template <index_t DIM, class T>
472  inline vecng<DIM, T> normalize(
473  const vecng<DIM, T>& v
474  ) {
475  T s = length(v);
476  if(s > 1e-30) {
477  s = T(1) / s;
478  }
479  return s * v;
480  }
