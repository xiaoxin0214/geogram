743  void draw_vertex(index_t v) {
744  if(do_animation_) {
745  if(mesh_->vertices.single_precision()) {
746  const GLUPfloat* p =
747  mesh_->vertices.single_precision_point_ptr(v);
748  float t = float(time_);
749  float s = 1.0f - float(time_);
750  glupPrivateVertex3f(
751  s*p[0] + t*p[3],
752  s*p[1] + t*p[4],
753  s*p[2] + t*p[5]
754  );
755  } else {
756  const GLUPdouble* p = mesh_->vertices.point_ptr(v);
757  double s = 1.0 - time_;
758  glupPrivateVertex3d(
759  s*p[0] + time_*p[3],
760  s*p[1] + time_*p[4],
761  s*p[2] + time_*p[5]
762  );
763  }
764  } else {
765  if(mesh_->vertices.single_precision()) {
766  if(mesh_->vertices.dimension() < 3) {
767  glupPrivateVertex2fv(
768  mesh_->vertices.single_precision_point_ptr(v)
769  );
770  } else {
771  glupPrivateVertex3fv(
772  mesh_->vertices.single_precision_point_ptr(v)
773  );
774  }
775  } else {
776  if(mesh_->vertices.dimension() < 3) {
777  glupPrivateVertex2dv(

1021  void draw_sequences(
1022  const MeshSubElementsStore& elements,
1023  std::function<void(index_t, index_t)> draw
1024  ) {
1025  // GLUP hardware primitive filtering.
1026  // Not supported in all profiles.
1027  // If supported, primitives are filtered by the GPU,
1028  // else, they are filtered here explicitly.
1029  const bool hw_filtering = hw_filtering_supported();
1030 
1031  Filter* filter = nullptr;
1032  if(&elements == &mesh_->vertices) {
1033  filter = &vertices_filter_;
1034  } else if(&elements == &mesh_->facets) {
1035  filter = &facets_filter_;
1036  } else if(&elements == &mesh_->cells) {
1037  filter = &cells_filter_;
1038  }
1039  geo_assert(filter != nullptr) ;
1040 
1041  filter->begin(elements.attributes(), hw_filtering);
1042  index_t e = 0;
1043  while(e < elements.nb()) {
1044  while(e<elements.nb() && !(hw_filtering||filter->test(e))) {
1045  ++e;
1046  }
1047  if(e<elements.nb()) {
1048  index_t begin = e;
1049  while(e<elements.nb() && (hw_filtering||filter->test(e))) {
1050  ++e;
1051  }
1052  index_t end = e;
1053  glupBasePickingId(GLUPuint64(begin));
1054  draw(begin,end);
1055  }
1056  }
1057  filter->end();
1058  glupBasePickingId(0);
1059  }
